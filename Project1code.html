# Take the existing "Repayment_Status" column and apply a function to each value.

df["Default"] = df["Repayment_Status"].apply(
    
    # The lambda function checks each value (x) in the Repayment_Status column.
    # If the value is "Defaulted", it assigns a 1, if not it assigns a 0.
    lambda x: 1 if x == "Defaulted" else 0
)

# "Default" is now 1s and 0s:
# 1 represents a default
# 0 represents no default
overall_default_rate = df["Default"].mean()

print("Overall Default Rate:", overall_default_rate)

# Create age groups into 4 groups
# pd.qcut a quantile-based discretization function used to divide data into equal-sized bins based on rank or sample quantiles
df["Age_Group"] = pd.qcut(df["Customer_Age"], 4)

# For each age group, calculate the average of the "Default" column.
# taking the mean gives the default rate for each age group.
age_default_rate = df.groupby("Age_Group")["Default"].mean()

# Creates bar chart based on age ranges and default rate
print("Default Rate by Age Group:")
print(age_default_rate)
